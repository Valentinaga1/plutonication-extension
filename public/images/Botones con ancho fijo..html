javascript
function downloadWallet(platform: string) {
    let storeLink = '';
    if (platform === 'android') {
        storeLink = 'https://play.google.com/store/apps/details?id=com.wallet.android';
    } else if (platform === 'ios') {
        storeLink = 'https://apps.apple.com/us/app/wallet/id123456789';

    // Generar el código QR con el enlace de descarga

    // No redirigir inmediatamente, el código QR permanecerá visible
}

function generateQRCode(inputText: string) {
    const qrCodeContainer = document.getElementById('qr-code');

    try {
        QRCode.toCanvas(qrCodeContainer, inputText, function(error) {
            if (error) {
                console.error('Something went wrong generating the QR code:', error);
            } else {
                console.log("QR code generated successfully");
                // Muestra el contenedor del código QR
                qrCodeContainer.style.display = 'block';
            }
        });
    } catch (error) {
        console.error('Something went wrong generating the QR code:', error);
    }
}

En este código, la función downloadWallet ahora solo genera el código QR, pero no redirige al usuario de inmediato. Luego, en la función generateQRCode, se muestra el contenedor del código QR una vez que se ha generado el código QR exitosamente. Luego, el usuario puede escanear el código QR para ser redirigido a la página de descarga.
